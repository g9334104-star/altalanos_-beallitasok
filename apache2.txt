# b) Telepítés
sudo apt update
sudo apt install mc apache2 git -y

# c) Git klón
git clone https://github.com/viraghkrisztian-pataky/Linux-vizsga.git

# d) html mappa átmásolása /web mappába
sudo mkdir -p /web
sudo cp -r ~/Linux-vizsga/html /web/

# e) Tulajdonos és csoport beállítása
sudo chown -R www-data:www-data /web/html/

# f) Jogosultságok (RWX / R-X / R-X = 755)
sudo chmod -R 755 /web/html/

# g/1) DocumentRoot átállítása
sudo nano /etc/apache2/sites-available/000-default.conf
# → DocumentRoot /web/html

# g/2) Directory blokk hozzáadása ugyanabba a fájlba
# <Directory /web/html>
#     Options Indexes FollowSymLinks
#     AllowOverride None
#     Require all granted
# </Directory>

# g/3) apache2.conf – /web mappa engedélyezése
sudo nano /etc/apache2/apache2.conf
# → <Directory /web/> blokkot add hozzá:
# <Directory /web/>
#     Options Indexes FollowSymLinks
#     AllowOverride None
#     Require all granted
# </Directory>

# g/4) DirectoryIndex – web.html legyen a főoldal
sudo nano /etc/apache2/mods-enabled/dir.conf
# → DirectoryIndex web.html index.html index.php

# g/5) /web mappa elérhetősége
sudo chmod 755 /web

# Apache újraindítás
sudo systemctl restart apache2
